148 [[nodiscard]] inline Real DQDC (const Real& _ce, const Real& _temperature = 0) const
149 {
150 const Real PERC (1e-06);
151
152 if (_ce > 0.0)
153 {
154 return 0.5 * ( Qe(_ce * (1 + PERC), _temperature) -
155 Qe(_ce * (1 - PERC), _temperature))
156 / (_ce * PERC);
157 }
158
159 return (Qe(PERC, _temperature) - Qe(0, _temperature)) / PERC;
160 }
